#1 RDD operations first 4 lines of the text file
from pyspark.sql import SparkSession

spark = SparkSession.builder \
    .appName("RDD Example - Show First 4 Lines") \
    .getOrCreate()

sc = spark.sparkContext

rdd = sc.textFile("example.txt")

first_four_lines = rdd.take(4)

print("First 4 lines of the file:")
for line in first_four_lines:
    print(line)

spark.stop()
