#4 String Functions and array function
from pyspark.sql import SparkSession
from pyspark.sql.functions import col, upper, lower, length, concat, substring
from pyspark.sql.functions import split, array, array_contains, explode, size

spark = SparkSession.builder.appName("StringArrayFunctionsDemo").getOrCreate()

data = [
    (1, "Alice Johnson", "HR,Admin"),
    (2, "Bob Smith", "IT,Support"),
    (3, "Cathy Brown", "Finance,Audit")
]

columns = ["id", "name", "skills"]

df = spark.createDataFrame(data, columns)

df.show(truncate=False)

df.withColumn("name_upper", upper(col("name"))).show()
df.withColumn("name_lower", lower(col("name"))).show()
df.withColumn("name_length", length(col("name"))).show()
df.withColumn("concat_example", concat(col("name"), col("skills"))).show()
df.withColumn("short_name", substring(col("name"), 1, 5)).show()

df_array = df.withColumn("skills_array", split(col("skills"), ","))
df_array.show(truncate=False)

df.withColumn("sample_array", 
              array(col("id").cast("string"), col("name"))
             ).show(truncate=False)

df_array.withColumn("has_IT", array_contains(col("skills_array"), "IT")).show()
df_array.withColumn("skills_count", size(col("skills_array"))).show()
df_array.select("id", explode(col("skills_array")).alias("skill")).show()

spark.stop()
